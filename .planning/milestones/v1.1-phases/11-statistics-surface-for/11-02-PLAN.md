```dart
/// STATSUI-01 / D-05 — per-day soul row pull for PTVF daily fold.
/// Returns one row per soul transaction (no SQL aggregation — fold happens
/// in Dart at `GetDailyJoyPerYenUseCase`).
/// HAPPY-05 — uses `_soulExpenseFilter` constant; survival rows excluded.
Future<List<DailySoulRowSampleWithDay>> getDailySoulRowsForPtvf({
  required String bookId,
  required DateTime startDate,
  required DateTime endDate,
}) async {
  final results = await _db
      .customSelect(
        "SELECT DATE(timestamp, 'unixepoch', 'localtime') as day, "
        'amount, soul_satisfaction '
        'FROM transactions '
        'WHERE book_id = ? AND $_soulExpenseFilter '
        'AND timestamp >= ? AND timestamp <= ? '
        'ORDER BY timestamp ASC',
        variables: [
          Variable.withString(bookId),
          Variable.withDateTime(startDate),
          Variable.withDateTime(endDate),
        ],
      )
      .get();

  return results
      .map(
        (row) => DailySoulRowSampleWithDay(
          day: DateTime.parse(row.read<String>('day')),
          amount: row.read<int>('amount'),
          soulSatisfaction: row.read<int>('soul_satisfaction'),
        ),
      )
      .toList();
}
```

(b) Add `getLargestMonthlyExpense` method (mirror `getBestJoyMoment` at lines 335-366 — but with TOTAL-ledger filter, NOT `_soulExpenseFilter`, per D-15):

```dart
/// STATSUI-06 / D-15 — single largest monthly expense for 物語 group 総 card.
/// TOTAL ledger filter (`is_deleted = 0 AND type = 'expense'`) — survival rows
/// included per D-15 SCOPE simplification (生存 spending rolls up into 総帳本).
/// Tiebreak: amount DESC, timestamp DESC.
Future<LargestMonthlyExpense?> getLargestMonthlyExpense({
  required String bookId,
  required DateTime startDate,
  required DateTime endDate,
}) async {
  final results = await _db
      .customSelect(
        'SELECT id, amount, category_id, timestamp '
        'FROM transactions '
        "WHERE book_id = ? AND is_deleted = 0 AND type = 'expense' "
        'AND timestamp >= ? AND timestamp <= ? '
        'ORDER BY amount DESC, timestamp DESC '
        'LIMIT 1',
        variables: [
          Variable.withString(bookId),
          Variable.withDateTime(startDate),
          Variable.withDateTime(endDate),
        ],
      )
      .get();

  if (results.isEmpty) return null;

  final row = results.first;
  return LargestMonthlyExpense(
    transactionId: row.read<String>('id'),
    amount: row.read<int>('amount'),
    categoryId: row.read<String>('category_id'),
    timestamp: row.read<DateTime>('timestamp'),
  );
}
```
