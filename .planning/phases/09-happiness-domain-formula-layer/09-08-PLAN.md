    ```dart
    import 'package:flutter_riverpod/flutter_riverpod.dart';
    import 'package:riverpod_annotation/riverpod_annotation.dart';

    import '../../../../features/analytics/domain/models/best_joy_moment_row.dart';
    import '../../../../features/analytics/domain/models/family_happiness.dart';
    import '../../../../features/analytics/domain/models/happiness_report.dart';
    import '../../../../features/analytics/domain/models/metric_result.dart';
    import '../../../../features/family_sync/presentation/providers/state_active_group.dart';
    import '../../../../features/home/presentation/providers/state_shadow_books.dart';
    import 'repository_providers.dart';

    part 'state_happiness.g.dart';

    /// HAPPY-01..04 personal happiness report.
    /// Resolves currency from the active book; passes (bookId, year, month, currencyCode)
    /// to GetHappinessReportUseCase.
    @riverpod
    Future<HappinessReport> happinessReport(
      Ref ref, {
      required String bookId,
      required int year,
      required int month,
      required String currencyCode,
    }) async {
      final useCase = ref.watch(getHappinessReportUseCaseProvider);
      return useCase.execute(
        bookId: bookId,
        year: year,
        month: month,
        currencyCode: currencyCode,
      );
    }

    /// HAPPY-04 standalone Top Joy.
    @riverpod
    Future<MetricResult<BestJoyMomentRow>> bestJoyMoment(
      Ref ref, {
      required String bookId,
      required int year,
      required int month,
    }) async {
      final useCase = ref.watch(getBestJoyMomentUseCaseProvider);
      return useCase.execute(bookId: bookId, year: year, month: month);
    }

    /// FAMILY-01..02 family happiness aggregate.
    /// Resolves shadowBooksProvider (D-09) and short-circuits when no group is active.
    @riverpod
    Future<FamilyHappiness> familyHappiness(
      Ref ref, {
      required int year,
      required int month,
    }) async {
      final activeGroup = ref.watch(activeGroupProvider).valueOrNull;
      if (activeGroup == null) {
        return FamilyHappiness(
          year: year, month: month, totalGroupSoulTx: 0,
          familyHighlightsSum: const Empty(),
          sharedJoyInsight: const Empty(),
          medianSatisfaction: const Empty(),
        );
      }

      final shadowBooks = await ref.watch(shadowBooksProvider.future);
      // RESEARCH Q6c open question: include current device's own book?
      // Per D-09: presentation passes List<String> bookIds. shadowBooksProvider returns
      // OTHER members' books. The current user's own book(s) must also be included
      // for "family" to mean "everyone". Resolve current book and prepend.
      // For Phase 9 contract: pass shadowBooks ONLY; Phase 10/11 may extend if a
      // separate currentBookId resolution is required. (Document this caveat.)
      final groupBookIds = shadowBooks.map((s) => s.book.id).toList();

      final useCase = ref.watch(getFamilyHappinessUseCaseProvider);
      return useCase.execute(
        groupBookIds: groupBookIds,
        year: year,
        month: month,
      );
    }
    ```

    Note on the open question (RESEARCH Q6c): The use case correctly handles `groupBookIds: []` by returning an empty FamilyHappiness — so even if the resolution here is incomplete, no crash will occur. Phase 10 (HomePage rebuild) will likely refine the bookId resolution as it needs that path anyway. Document this in the SUMMARY.
