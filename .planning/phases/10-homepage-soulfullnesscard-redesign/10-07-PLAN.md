/// Integrated hero card on the HomePage replacing MonthOverviewCard + LedgerComparisonSection + SoulFullnessCard (Phase 10).
///
/// **Pure StatelessWidget** — no Riverpod refs inside; parent (`home_screen.dart`)
/// resolves all `AsyncValue.when()` and passes resolved Freezed aggregates.
/// (UI-SPEC line 277; CLAUDE.md Widget Parameter Pattern.)
///
/// Mode discrimination is a top-level branch via `isGroupMode`:
///   - single mode: rings encode [HappinessReport]; family region is `SizedBox.shrink()`
///   - group mode + non-empty `shadowBooks`: rings encode [FamilyHappiness]; member rows render
///   - group mode + empty `shadowBooks`: rings still render group-mode metrics; member section collapses (D-08)
///
/// Hard contracts (CONTEXT.md D-01..D-13, RESEARCH §"Pitfalls"):
///   - All ¥ amounts use `AppTextStyles.amount*` (tabular figures); never hardcode `'JPY'`.
///   - Sealed `MetricResult<T>` consumed via `switch` only — never `as Value` casts.
///   - Exactly 2 `Icons.info_outline` instances total (HOMEUI-04).
///   - Whole-card single `onTap`; sub-widget taps bubble up; ⓘ icons absorb taps (Pitfall #3).
///   - 魂/生存 split bar shows ABSOLUTE amounts; never frame as joy ROI / share / ratio (D-02).
///   - No streak / badge / target / cross-period happiness chips (ADR-012 binding).
class HomeHeroCard extends StatelessWidget {
  const HomeHeroCard({
    required this.report,
    required this.happiness,
    required this.bestJoy,
    required this.family,
    required this.shadowBooks,
    required this.shadowAggregate,
    required this.currencyCode,
    required this.locale,
    required this.isGroupMode,
    required this.onTap,
    super.key,
  });

  final MonthlyReport report;
  final HappinessReport happiness;
  final MetricResult<BestJoyMomentRow> bestJoy;
  final FamilyHappiness? family;          // null in single mode
  final List<ShadowBookInfo>? shadowBooks; // null in single mode
  final ShadowAggregate? shadowAggregate;  // null in single mode
  final String currencyCode;               // from Book.currency via bookByIdProvider; never literal 'JPY' here
  final Locale locale;
  final bool isGroupMode;
  final VoidCallback onTap;

  @override
  Widget build(BuildContext context) {
    final l10n = S.of(context);

    return GestureDetector(
      onTap: onTap,
      behavior: HitTestBehavior.opaque,
      child: Container(
        padding: const EdgeInsets.all(18),    // UI-SPEC OQ #1 — 18 picked for hero weight
        decoration: BoxDecoration(
          color: context.wmCard,
          borderRadius: BorderRadius.circular(14),
          border: Border.all(color: context.wmBorderDefault),
        ),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            _buildHeroHeader(context, l10n),
            const SizedBox(height: 16),
            _buildSplitBar(context, l10n),
            const SizedBox(height: 12),
            _buildDivider(context),
            const SizedBox(height: 12),
            _buildRingSection(context, l10n),
            const SizedBox(height: 12),
            _buildDivider(context),
            const SizedBox(height: 12),
            _buildBestJoyStrip(context, l10n),
            if (isGroupMode && (shadowBooks?.isNotEmpty ?? false)) ...[
              const SizedBox(height: 12),
              _buildDivider(context),
              const SizedBox(height: 12),
              _buildMembersSection(context, l10n),
            ],
          ],
        ),
      ),
    );
  }

  // ─── Region 1: Hero header ─────────────────────────────────────────────────

  Widget _buildHeroHeader(BuildContext context, S l10n) {
    // Group-mode total = own + shadow totals; single mode = own only.
    final totalForDisplay = isGroupMode
        ? report.totalExpenses + (shadowAggregate?.totalExpenses ?? 0)
        : report.totalExpenses;
    final prevForDisplay = isGroupMode
        ? (report.previousMonthComparison?.previousExpenses ?? 0) + (shadowAggregate?.prevTotalExpenses ?? 0)
        : (report.previousMonthComparison?.previousExpenses ?? 0);

    const formatter = FormatterService();
    final totalText = formatter.formatCurrency(totalForDisplay, currencyCode, locale);
    final prevText = formatter.formatCurrency(prevForDisplay, currencyCode, locale);

    final label = isGroupMode ? l10n.homeHeroCardLabelGroup : l10n.homeHeroCardLabelSingle;

    // Trend chip — hidden when totalForDisplay == 0 AND prevForDisplay == 0 (D-09 empty state).
    final showTrendChip = !(totalForDisplay == 0 && prevForDisplay == 0);
    final trendPercent = prevForDisplay > 0
        ? ((totalForDisplay - prevForDisplay) / prevForDisplay * 100).round()
        : 0;

    return Row(
      crossAxisAlignment: CrossAxisAlignment.center,
      children: [
        Expanded(
          child: Column(
            crossAxisAlignment: CrossAxisAlignment.start,
            children: [
              Text(
                label,
                style: AppTextStyles.bodyMedium.copyWith(color: context.wmTextSecondary),
              ),
              const SizedBox(height: 4),
              Text(
                totalText,
                style: AppTextStyles.amountLarge.copyWith(color: context.wmTextPrimary),
              ),
              const SizedBox(height: 4),
              Text(
                l10n.homeHeroPreviousMonthSubline(prevText),
                style: AppTextStyles.bodyMedium.copyWith(color: context.wmTextSecondary),
              ),
            ],
          ),
        ),
        if (showTrendChip) _buildTrendChip(context, trendPercent),
      ],
    );
  }

  Widget _buildTrendChip(BuildContext context, int trendPercent) {
    // Reuses pattern from MonthOverviewCard:79-104 verbatim.
    final isUp = trendPercent > 0;
    final isFlatOrDown = trendPercent <= 0;
    final trendText = isFlatOrDown ? '$trendPercent%' : '+$trendPercent%';
    return Container(
      padding: const EdgeInsets.symmetric(horizontal: 10, vertical: 4),
      decoration: BoxDecoration(
        color: AppColors.oliveLight,
        borderRadius: BorderRadius.circular(999),
      ),
      child: Row(
        mainAxisSize: MainAxisSize.min,
        children: [
          Icon(
            isUp ? Icons.trending_up : Icons.trending_down,
            size: 14,
            color: AppColors.olive,
          ),
          const SizedBox(width: 4),
          Text(
            trendText,
            style: AppTextStyles.bodySmall.copyWith(
              fontWeight: FontWeight.w600,
              color: AppColors.olive,
            ),
          ),
        ],
      ),
    );
  }

  // ─── Region 2: 魂/生存 split bar ────────────────────────────────────────────

      ],
    );
  }

  Widget _buildDivider(BuildContext context) {

  }

  // ─── Region 4: Ring section ───────────────────────────────────────────────

  Widget _buildRingSection(BuildContext context, S l10n) {

  HappinessRingsPainter _buildRingsPainter(BuildContext context) {
    // Single mode encodes HappinessReport (D-03);
    // Group mode encodes FamilyHappiness (D-04).
    // Sweep ratio computation uses the helpers below (Pattern 4 from RESEARCH).
    if (!isGroupMode) {
      return HappinessRingsPainter(
        outerSweepRatio: _outerSweepRatioSingle(happiness.joyPerYen),
        middleSweepRatio: _middleSweepRatioSingle(happiness.avgSatisfaction),
        innerSweepRatio: _innerSweepRatioSingle(happiness.highlightsCount, happiness.totalSoulTx),
        outerGradient: SweepGradient(colors: [AppColors.soul.withValues(alpha: 0.6), AppColors.soul]),
        middleGradient: SweepGradient(colors: [AppColors.olive.withValues(alpha: 0.6), AppColors.olive]),
        innerGradient: SweepGradient(colors: [AppColors.accentPrimary.withValues(alpha: 0.6), AppColors.accentPrimary]),
        trackColor: context.wmBackgroundDivider,
      );
    }
    // Group mode
    final fam = family;
    return HappinessRingsPainter(
      outerSweepRatio: fam == null ? null : _outerSweepRatioGroup(fam.familyHighlightsSum),
      middleSweepRatio: fam == null ? null : _middleSweepRatioGroup(fam.sharedJoyInsight),
      innerSweepRatio: fam == null ? null : _innerSweepRatioGroup(fam.medianSatisfaction),
      outerGradient: SweepGradient(colors: [AppColors.shared.withValues(alpha: 0.6), AppColors.shared]),
      middleGradient: SweepGradient(colors: [AppColors.accentPrimary.withValues(alpha: 0.6), AppColors.accentPrimary]),
      innerGradient: SweepGradient(colors: [AppColors.olive.withValues(alpha: 0.6), AppColors.olive]),
      trackColor: context.wmBackgroundDivider,
    );
  }

  /// Single mode outer ring — joyPerYen normalized 0..2 (Pattern 4 fallback per A7).
  /// Phase 10 ships fallback only; Phase 11 may add prev-month query.
  double? _outerSweepRatioSingle(MetricResult<double> joyPerYen) {
    return switch (joyPerYen) {
      Empty() => null,
      Value(:final data) => (data / 2.0).clamp(0.0, 1.0),
    };
  }

  double? _middleSweepRatioSingle(MetricResult<double> avgSat) {
    return switch (avgSat) {
      Empty() => null,
      Value(:final data) => (data / 10.0).clamp(0.0, 1.0),    // sat is 1-10 → ratio 0..1
    };
  }

  double? _innerSweepRatioSingle(MetricResult<int> highlights, int totalSoulTx) {
    return switch (highlights) {
      Empty() => null,
      Value(:final data) => totalSoulTx > 0 ? (data / totalSoulTx).clamp(0.0, 1.0) : null,
    };
  }

  double? _outerSweepRatioGroup(MetricResult<int> familyHighlightsSum) {
    return switch (familyHighlightsSum) {
      Empty() => null,
      Value(:final data) => (data / 30.0).clamp(0.0, 1.0),    // /30 — reasonable monthly ceiling
    };
  }

  double? _middleSweepRatioGroup(MetricResult<dynamic> sharedJoyInsight) {
    // Binary: present (full sweep) vs Empty (track only)
    return switch (sharedJoyInsight) {
      Empty() => null,
      Value(_) => 1.0,
    };
  }

  double? _innerSweepRatioGroup(MetricResult<double> medianSat) {
    return switch (medianSat) {
      Empty() => null,
      Value(:final data) => (data / 10.0).clamp(0.0, 1.0),
    };
  }

  String _buildCenterText() {
    // Single: avgSatisfaction value (e.g. "7.8") or "—" on Empty
    // Group: familyHighlightsSum int (e.g. "27") or "—" on Empty
    if (!isGroupMode) {
      return switch (happiness.avgSatisfaction) {
        Empty() => '—',
        Value(:final data) => data.toStringAsFixed(1),
      };
    }
    final fam = family;
    if (fam == null) return '—';
    return switch (fam.familyHighlightsSum) {
      Empty() => '—',
      Value(:final data) => '$data',
    };
  }
