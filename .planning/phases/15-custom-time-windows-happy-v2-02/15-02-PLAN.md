```dart
import 'package:freezed_annotation/freezed_annotation.dart';
part 'time_window.freezed.dart';

/// Time window for AnalyticsScreen Joy-metric queries (HAPPY-V2-02).
/// All variants resolve to inclusive `(start 00:00:00, end 23:59:59)` per D-06.
/// Week starts Monday for all locales per D-05.
@freezed
sealed class TimeWindow with _$TimeWindow {
  const TimeWindow._();

  /// ISO-week-Monday-anchored 7-day window. [mondayStart] MUST be a Monday;
  /// constructor callers are responsible for snapping to Monday (D-05).
  const factory TimeWindow.week({required DateTime mondayStart}) = WeekWindow;

  /// Calendar month: [year]-[month] (month in 1..12).
  const factory TimeWindow.month({required int year, required int month}) = MonthWindow;

  /// Calendar quarter (1..4). Q1=Jan-Mar, Q2=Apr-Jun, Q3=Jul-Sep, Q4=Oct-Dec.
  const factory TimeWindow.quarter({required int year, required int quarter}) = QuarterWindow;

  /// Calendar year.
  const factory TimeWindow.year({required int year}) = YearWindow;

  /// User-picked arbitrary range. Validation (start <= end, ≤ 12 months,
  /// end ≤ today) lives in `TimeWindowValidation.assertValid` (application).
  const factory TimeWindow.custom({
    required DateTime startDate,
    required DateTime endDate,
  }) = CustomWindow;
}

extension TimeWindowRange on TimeWindow {
  /// Returns the inclusive `(start, end)` pair for this window per D-06.
  /// start = midnight of the first day; end = 23:59:59 of the last day.
  ({DateTime start, DateTime end}) get range => switch (this) {
        WeekWindow(:final mondayStart) => (
            start: DateTime(mondayStart.year, mondayStart.month, mondayStart.day),
            end: DateTime(mondayStart.year, mondayStart.month, mondayStart.day + 6, 23, 59, 59),
          ),
        MonthWindow(:final year, :final month) => (
            start: DateTime(year, month, 1),
            end: DateTime(year, month + 1, 0, 23, 59, 59),
          ),
        QuarterWindow(:final year, :final quarter) => (
            start: DateTime(year, (quarter - 1) * 3 + 1, 1),
            end: DateTime(year, quarter * 3 + 1, 0, 23, 59, 59),
          ),
        YearWindow(:final year) => (
            start: DateTime(year, 1, 1),
            end: DateTime(year, 12, 31, 23, 59, 59),
          ),
        CustomWindow(:final startDate, :final endDate) => (
            start: DateTime(startDate.year, startDate.month, startDate.day),
            end: DateTime(endDate.year, endDate.month, endDate.day, 23, 59, 59),
          ),
      };
}
```

```dart
/// D-06 + D-07 + D-08 — defensive validation invoked at the use-case boundary.
/// UI surfaces localized SnackBar BEFORE calling the use case; this helper
/// is defense-in-depth against bad input reaching the repository.
class TimeWindowValidation {
  TimeWindowValidation._();

  /// Throws ArgumentError if start > end, span > 12 months, or end > now.
  /// Uses calendar-month math for the 12-month gate (NOT Duration.inDays —
  /// Duration is leap-year off-by-one per 15-RESEARCH.md Pitfall #5).
  static void assertValid(DateTime startDate, DateTime endDate) {
    if (startDate.isAfter(endDate)) {
      throw ArgumentError.value((startDate, endDate), 'window', 'startDate must be <= endDate');
    }
    final months = (endDate.year - startDate.year) * 12 + (endDate.month - startDate.month);
    if (months > 12 || (months == 12 && endDate.day > startDate.day)) {
      throw ArgumentError.value((startDate, endDate), 'window', 'window must not exceed 12 months');
    }
    if (endDate.isAfter(DateTime.now())) {
      throw ArgumentError.value((startDate, endDate), 'window', 'endDate must not be in the future');
    }
  }
}
```
