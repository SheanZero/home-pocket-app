```dart
import 'package:fl_chart/fl_chart.dart';
import 'package:flutter/material.dart';

import '../../../../application/i18n/formatter_service.dart';
import '../../../../core/theme/app_colors.dart';
import '../../../../core/theme/app_text_styles.dart';
import '../../../../core/theme/app_theme_colors.dart';
import '../../../../generated/l10n.dart' show S;
import '../../domain/models/expense_trend.dart';

/// STATSUI-06 — 6 か月支出推移 BarChart (時間 group · 総 card).
/// Current month bar highlighted via thicker stroke + survival fill;
/// other bars use lighter survival tint.
class MonthlySpendTrendBarChart extends StatelessWidget {
  const MonthlySpendTrendBarChart({
    super.key,
    required this.trendData,
    required this.selectedYear,
    required this.selectedMonth,
    required this.locale,
  });

  final ExpenseTrendData trendData;
  final int selectedYear;
  final int selectedMonth;
  final Locale locale;

  @override
  Widget build(BuildContext context) {
    final l10n = S.of(context);
    final formatter = FormatterService();
    final months = trendData.months;
    if (months.isEmpty) {
      return const SizedBox.shrink();
    }
    final maxY = months.fold<double>(
      0,
      (acc, m) => m.total > acc ? m.total : acc,
    );

    return SizedBox(
      height: 200,
      child: BarChart(
        BarChartData(
          maxY: maxY * 1.15,
          minY: 0,
          alignment: BarChartAlignment.spaceAround,
          gridData: FlGridData(
            show: true,
            drawVerticalLine: false,
            horizontalInterval: maxY > 0 ? maxY / 4 : 1,
          ),
          borderData: FlBorderData(show: false),
          titlesData: FlTitlesData(
            topTitles: const AxisTitles(sideTitles: SideTitles(showTitles: false)),
            rightTitles: const AxisTitles(sideTitles: SideTitles(showTitles: false)),
            bottomTitles: AxisTitles(
              sideTitles: SideTitles(
                showTitles: true,
                getTitlesWidget: (value, meta) {
                  final idx = value.toInt();
                  if (idx < 0 || idx >= months.length) {
                    return const SizedBox.shrink();
                  }
                  final m = months[idx];
                  return Padding(
                    padding: const EdgeInsets.only(top: 4),
                    child: Text(
                      '${m.month}月',
                      style: AppTextStyles.amountSmall.copyWith(
                        color: context.wmTextSecondary,
                      ),
                    ),
                  );
                },
              ),
            ),
            leftTitles: AxisTitles(
              sideTitles: SideTitles(
                showTitles: true,
                reservedSize: 50,
                interval: maxY > 0 ? maxY / 4 : 1,
                getTitlesWidget: (value, meta) {
                  if (value == 0) return const SizedBox.shrink();
                  return Text(
                    formatter.formatCompact(value, locale),
                    style: AppTextStyles.amountSmall.copyWith(
                      color: context.wmTextSecondary,
                    ),
                  );
                },
              ),
            ),
          ),
          barGroups: months.asMap().entries.map((entry) {
            final i = entry.key;
            final m = entry.value;
            final isCurrent =
                m.year == selectedYear && m.month == selectedMonth;
            return BarChartGroupData(
              x: i,
              barRods: [
                BarChartRodData(
                  toY: m.total,
                  color: isCurrent
                      ? AppColors.survival
                      : AppColors.survival.withValues(alpha: 0.30),
                  borderSide: isCurrent
                      ? const BorderSide(color: Colors.transparent, width: 0)
                      : BorderSide.none,
                  width: 18,
                  borderRadius: const BorderRadius.only(
                    topLeft: Radius.circular(4),
                    topRight: Radius.circular(4),
                  ),
                ),
              ],
            );
          }).toList(),
          barTouchData: BarTouchData(
            touchTooltipData: BarTouchTooltipData(
              getTooltipItem: (group, _, rod, __) {
                final m = months[group.x];
                return BarTooltipItem(
                  '${m.year}/${m.month}\n${formatter.formatCompact(rod.toY, locale)}',
                  AppTextStyles.amountSmall,
                );
              },
            ),
          ),
        ),
      ),
    );
  }
}
```

```dart
import 'package:fl_chart/fl_chart.dart';
import 'package:flutter/material.dart';

import '../../../../core/theme/app_colors.dart';
import '../../../../core/theme/app_text_styles.dart';
import '../../../../core/theme/app_theme_colors.dart';
import '../../../../generated/l10n.dart' show S;
import '../../../../infrastructure/i18n/formatters/joy_density_formatter.dart';
import '../../domain/models/daily_joy_per_yen_point.dart';
import '../../domain/models/metric_result.dart';

/// STATSUI-01 — Joy/¥ MTD LineChart with gap-vs-zero (D-06).
/// Days without soul tx render as line GAPS, not zero points.
/// Multiple `LineChartBarData` entries — one per contiguous segment.
/// Baseline-anchored Y-axis (minY=0).
class JoyTrendLineChart extends StatelessWidget {
  const JoyTrendLineChart({
    super.key,
    required this.result,
    required this.daysInMonth,
    required this.currencyCode,
    required this.locale,
  });

  final MetricResult<List<DailyJoyPerYenPoint>> result;
  final int daysInMonth;
  final String currencyCode;
  final Locale locale;

  @override
  Widget build(BuildContext context) {
    return switch (result) {
      Empty() => const SizedBox(height: 180),
      Value(:final data) => _buildChart(context, data as List<DailyJoyPerYenPoint>),
    };
  }

  Widget _buildChart(BuildContext context, List<DailyJoyPerYenPoint> points) {
    if (points.isEmpty) return const SizedBox(height: 180);
    final maxY = points.fold<double>(
      0,
      (acc, p) => p.joyPerYen > acc ? p.joyPerYen : acc,
    );
    final segments = _splitIntoContiguousSegments(points, daysInMonth);

    return SizedBox(
      height: 180,
      child: LineChart(
        LineChartData(
          minX: 1,
          maxX: daysInMonth.toDouble(),
          minY: 0, // D-06: baseline-anchored — never auto-zoom into negative
          maxY: maxY * 1.20,
          gridData: FlGridData(
            show: true,
            drawVerticalLine: false,
            horizontalInterval: maxY > 0 ? maxY / 4 : 1,
          ),
          borderData: FlBorderData(show: false),
          titlesData: FlTitlesData(
            topTitles: const AxisTitles(sideTitles: SideTitles(showTitles: false)),
            rightTitles: const AxisTitles(sideTitles: SideTitles(showTitles: false)),
            bottomTitles: AxisTitles(
              sideTitles: SideTitles(
                showTitles: true,
                interval: 5,
                getTitlesWidget: (value, _) {
                  final d = value.toInt();
                  if (d < 1 || d > daysInMonth) return const SizedBox.shrink();
                  return Text(
                    '$d',
                    style: AppTextStyles.amountSmall.copyWith(
                      color: context.wmTextSecondary,
                    ),
                  );
                },
              ),
            ),
            leftTitles: AxisTitles(
              sideTitles: SideTitles(
                showTitles: true,
                reservedSize: 50,
                interval: maxY > 0 ? maxY / 4 : 1,
                getTitlesWidget: (value, _) {
                  if (value == 0) return const SizedBox.shrink();
                  return Text(
                    // Pitfall 4: currency-aware Y-axis label, never 'JPY' literal
                    formatJoyDensity(value, currencyCode),
                    style: AppTextStyles.amountSmall.copyWith(
                      color: context.wmTextSecondary,
                    ),
                  );
                },
              ),
            ),
          ),
          lineBarsData: [
            for (final seg in segments)
              LineChartBarData(
                spots: seg
                    .map((p) => FlSpot(p.day.toDouble(), p.joyPerYen))
                    .toList(),
                color: AppColors.soul,
                barWidth: 3,
                isCurved: false, // straight segments preserve "gap" semantics
                dotData: const FlDotData(show: true),
              ),
          ],
        ),
      ),
    );
  }

  /// D-06: split points into contiguous-day runs.
  static List<List<DailyJoyPerYenPoint>> _splitIntoContiguousSegments(
    List<DailyJoyPerYenPoint> points,
    int monthDays,
  ) {
    final segments = <List<DailyJoyPerYenPoint>>[];
    final byDay = {for (final p in points) p.day: p};
    var current = <DailyJoyPerYenPoint>[];
    for (var d = 1; d <= monthDays; d++) {
      final point = byDay[d];
      if (point != null) {
        current.add(point);
      } else if (current.isNotEmpty) {
        segments.add(current);
        current = <DailyJoyPerYenPoint>[];
      }
    }
    if (current.isNotEmpty) segments.add(current);
    return segments;
  }
}
```

```dart
/// STATSUI-06 — 類別支出分布 (分布 group · 総 card).
/// Top-N + 'その他' bucket; donut style via centerSpaceRadius.
/// Legend Wrap below chart with category name + percent.
class CategorySpendDonutChart extends StatelessWidget {
  const CategorySpendDonutChart({
    super.key,
    required this.breakdowns,
    this.topN = 6,
  });

  final List<CategoryBreakdown> breakdowns;
  final int topN;

  static const List<Color> _palette = [
    AppColors.survival,
    AppColors.soul,
    AppColors.olive,
    AppColors.accentPrimary,
    // Mix in alpha-shifted survivals to fill 6-7 slots without inline hex
  ];

  @override
  Widget build(BuildContext context) {
    final l10n = S.of(context);
    if (breakdowns.isEmpty) {
      return Padding(
        padding: const EdgeInsets.all(16),
        child: Text(
          l10n.analyticsCardEmptyLargestExpense,
          style: AppTextStyles.bodyMedium.copyWith(
            color: context.wmTextSecondary,
          ),
        ),
      );
    }

    // breakdowns already ORDER BY total DESC at SQL — take first N + bucket rest
    final top = breakdowns.take(topN).toList();
    final restAmount = breakdowns
        .skip(topN)
        .fold<int>(0, (acc, b) => acc + b.amount);
    final restPercent = breakdowns
        .skip(topN)
        .fold<double>(0, (acc, b) => acc + b.percentage);
    final hasOther = restAmount > 0;

    final sections = <PieChartSectionData>[
      for (var i = 0; i < top.length; i++)
        PieChartSectionData(
          value: top[i].amount.toDouble(),
          title: '${top[i].percentage.toStringAsFixed(0)}%',
          color: _palette[i % _palette.length],
          radius: 70,
          titleStyle: AppTextStyles.amountSmall.copyWith(color: Colors.white),
        ),
      if (hasOther)
        PieChartSectionData(
          value: restAmount.toDouble(),
          title: '${restPercent.toStringAsFixed(0)}%',
          color: AppColors.olive.withValues(alpha: 0.4),
          radius: 70,
          titleStyle: AppTextStyles.amountSmall.copyWith(color: Colors.white),
        ),
    ];

    return Column(
      crossAxisAlignment: CrossAxisAlignment.start,
      children: [
        SizedBox(
          height: 180,
          child: PieChart(
            PieChartData(
              sections: sections,
              centerSpaceRadius: 35, // donut
              sectionsSpace: 2,
            ),
          ),
        ),
        const SizedBox(height: 8),
        Wrap(
          spacing: 16,
          runSpacing: 4,
          children: [
            for (var i = 0; i < top.length; i++)
              _LegendItem(
                color: _palette[i % _palette.length],
                label: top[i].categoryName,
              ),
            if (hasOther)
              _LegendItem(
                color: AppColors.olive.withValues(alpha: 0.4),
                label: l10n.analyticsCategoryDonutOther,
              ),
          ],
        ),
      ],
    );
  }
}
```

```dart
import 'package:fl_chart/fl_chart.dart';

/// STATSUI-02 — 満足度分布 1-10 BarChart (分布 group · 悦己 card).
/// - Per-bar cool→warm gradient (D-10) — sat 1 cool, sat 10 warm.
///   Starting hue = AppColors.survival (NOT red/orange) per ADR-014 guard.
/// - Bar 5 carries permanent annotation `analyticsHistogramBarFiveAnnotation`
///   (HARD-LOCKED trilingual). Per Pitfall 5 — bar 5 always rendered (1px stub
///   if score=5 absent) so the annotation is always visible.
/// - Persistent caption below chart shows ADR-014 ordinal-color guard.
/// - Per-bar Semantics reads neutral "satisfaction value + count + total" —
///   NEVER includes negative-emotion words in any locale (Pitfall 7).
class SatisfactionDistributionHistogram extends StatelessWidget {
  const SatisfactionDistributionHistogram({
    super.key,
    required this.distribution,
  });

  final List<SatisfactionDistributionResult> distribution;

  /// D-10 cool→warm gradient resolved via AppColors token interpolation.
  static List<Color> _gradientColors() {
    return [
      AppColors.survival, // 1 cool
      Color.lerp(AppColors.survival, AppColors.soul, 0.30)!,
      Color.lerp(AppColors.survival, AppColors.soul, 0.50)!,
      Color.lerp(AppColors.survival, AppColors.soul, 0.70)!,
      AppColors.soul, // 5 (median band)
      Color.lerp(AppColors.soul, AppColors.olive, 0.40)!,
      Color.lerp(AppColors.soul, AppColors.olive, 0.60)!,
      Color.lerp(AppColors.soul, AppColors.accentPrimary, 0.70)!,
      Color.lerp(AppColors.soul, AppColors.accentPrimary, 0.90)!,
      AppColors.accentPrimary, // 10 warm
    ];
  }

  @override
  Widget build(BuildContext context) {
    final l10n = S.of(context);
    final colors = _gradientColors();

    // Pitfall 5: zero-fill all 10 scores so bar 5 always renders.
    final byScore = {for (final r in distribution) r.score: r.count};
    final total = distribution.fold<int>(0, (acc, r) => acc + r.count);
    final maxCount = byScore.values.fold<int>(0, (m, c) => c > m ? c : m);

    return Column(
      crossAxisAlignment: CrossAxisAlignment.start,
      children: [
        SizedBox(
          height: 200,
          child: Stack(
            children: [
              BarChart(
                BarChartData(
                  maxY: (maxCount + 1).toDouble(),
                  minY: 0,
                  alignment: BarChartAlignment.spaceAround,
                  borderData: FlBorderData(show: false),
                  gridData: const FlGridData(show: false),
                  titlesData: FlTitlesData(
                    topTitles: const AxisTitles(sideTitles: SideTitles(showTitles: false)),
                    rightTitles: const AxisTitles(sideTitles: SideTitles(showTitles: false)),
                    leftTitles: AxisTitles(
                      sideTitles: SideTitles(
                        showTitles: true,
                        reservedSize: 30,
                        getTitlesWidget: (value, _) {
                          if (value == 0) return const SizedBox.shrink();
                          return Text(
                            value.toInt().toString(),
                            style: AppTextStyles.amountSmall.copyWith(
                              color: context.wmTextSecondary,
                            ),
                          );
                        },
                      ),
                    ),
                    bottomTitles: AxisTitles(
                      sideTitles: SideTitles(
                        showTitles: true,
                        getTitlesWidget: (value, _) {
                          final score = value.toInt() + 1;
                          return Padding(
                            padding: const EdgeInsets.only(top: 4),
                            child: Text(
                              '$score',
                              style: AppTextStyles.amountSmall.copyWith(
                                color: context.wmTextSecondary,
                              ),
                            ),
                          );
                        },
                      ),
                    ),
                  ),
                  barGroups: List.generate(10, (idx) {
                    final score = idx + 1;
                    final count = byScore[score] ?? 0;
                    // T-Information-2 + Pitfall 7: neutral semantic label only.
                    return BarChartGroupData(
                      x: idx,
                      barRods: [
                        BarChartRodData(
                          // Render 1px stub when count==0 so bar 5 annotation
                          // anchor is preserved (Pitfall 5 normalization).
                          toY: count == 0 ? 0.05 : count.toDouble(),
                          color: colors[idx],
                          width: 18,
                          borderRadius: const BorderRadius.only(
                            topLeft: Radius.circular(2),
                            topRight: Radius.circular(2),
                          ),
                        ),
                      ],
                    );
                  }),
                  barTouchData: BarTouchData(
                    touchTooltipData: BarTouchTooltipData(
                      getTooltipItem: (group, _, rod, __) {
                        final score = group.x + 1;
                        final count = byScore[score] ?? 0;
                        return BarTooltipItem(
                          '$score/10\n$count',
                          AppTextStyles.amountSmall,
                        );
                      },
                    ),
                  ),
                ),
              ),
              // Bar-5 annotation overlay (Pattern 3 — fl_chart 0.69 BarChartRodLabel
              // API may differ; Stack overlay is the universal fallback per
              // Assumption A1).
              Positioned(
                left: 0,
                right: 0,
                top: 0,
                child: _BarFiveAnnotation(
                  text: l10n.analyticsHistogramBarFiveAnnotation,
                ),
              ),
            ],
          ),
        ),
        const SizedBox(height: 6),
        // ADR-014 guard caption — D-10 binding.
        Text(
          l10n.analyticsHistogramColorCaption,
          style: AppTextStyles.caption.copyWith(
            color: context.wmTextSecondary,
          ),
        ),
        // Per-bar accessible region — neutral values only.
        Semantics(
          // T-Information-2 + Pitfall 7: never include negative-emotion words.
          label: _buildSemanticLabel(byScore, total),
          excludeSemantics: false,
          child: const SizedBox.shrink(),
        ),
      ],
    );
  }

  String _buildSemanticLabel(Map<int, int> byScore, int total) {
    final buf = StringBuffer();
    for (var s = 1; s <= 10; s++) {
      final c = byScore[s] ?? 0;
      buf.write('satisfaction $s of 10 — $c of $total. ');
    }
    return buf.toString().trim();
  }
}

class _BarFiveAnnotation extends StatelessWidget {
  const _BarFiveAnnotation({required this.text});
  final String text;

  @override
  Widget build(BuildContext context) {
    // Bar 5 is at index 4 of 10 evenly-spaced bars. Approximate horizontal
    // anchor at 45% width (idx 4 of 10 + spaceAround padding).
    return Align(
      alignment: const Alignment(-0.10, -1.0),
      child: Padding(
        padding: const EdgeInsets.symmetric(horizontal: 4, vertical: 2),
        child: Text(
          text,
          style: AppTextStyles.caption.copyWith(
            color: AppColors.soul,
            fontWeight: FontWeight.w700,
          ),
        ),
      ),
    );
  }
}
```
