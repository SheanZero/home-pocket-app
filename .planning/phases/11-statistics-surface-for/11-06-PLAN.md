class LargestExpenseStoryCard extends StatelessWidget {
  const LargestExpenseStoryCard({
    super.key,
    required this.expense,
    required this.currencyCode,
    required this.locale,
    this.onTap,
  });

  final LargestMonthlyExpense? expense;

          padding: const EdgeInsets.all(14),
          child: Column(
            crossAxisAlignment: CrossAxisAlignment.start,
            children: [
              Text(
                l10n.analyticsCardTitleLargestExpense,
                style: AppTextStyles.titleLarge.copyWith(
                  color: AppColors.survival,
                ),
              ),
              const SizedBox(height: 8),
              if (expense == null)
                Text(
                  l10n.analyticsCardEmptyLargestExpense,
                  style: AppTextStyles.bodyMedium.copyWith(
                    color: context.wmTextSecondary,
                  ),
                )
              else
                _buildBody(context, l10n, formatter, expense!),
            ],
          ),
        ),
      ),
    );

    BuildContext context,
    S l10n,
    FormatterService formatter,
    LargestMonthlyExpense e,
  ) {
    // CategoryLocalizationService.resolveFromId returns localized name —

    return Semantics(
      // T-Information-1 mitigation: label reads category + amount + date only.
      label: '$categoryName $amountText $dateLabel',
      child: Text(
        l10n.analyticsCardLargestExpenseBody(categoryName, amountText, dateLabel),
        style: AppTextStyles.bodyMedium.copyWith(color: context.wmTextPrimary),
      ),
    );
  }

import 'package:flutter/material.dart';

import '../../../../application/i18n/formatter_service.dart';
import '../../../../core/theme/app_colors.dart';
import '../../../../core/theme/app_text_styles.dart';
import '../../../../core/theme/app_theme_colors.dart';
import '../../../../generated/l10n.dart' show S;
import '../../../../infrastructure/i18n/formatters/date_formatter.dart';
import '../../domain/models/best_joy_moment_row.dart';
import '../../domain/models/metric_result.dart';

/// STATSUI-02 (anchor card in 物語 group · 悦己).
/// 3-arm switch: Empty / Value(sat<=2) → empty state (anti-`¥10 candy` per
/// Phase 10 D-04); Value(sat>2) → BIG `category · date` + SMALL `¥amount · 満足 sat/10 ✨`.
/// D-14 binding: SEPARATE widget from HomeHeroCard's _buildBestJoyStrip — no
/// shared base widget extraction (hero contract vs analytics-card contract).
class BestJoyStoryStrip extends StatelessWidget {
  const BestJoyStoryStrip({
    super.key,
    required this.bestJoy,
    required this.currencyCode,
    required this.locale,
    this.onTap,
  });

  final MetricResult<BestJoyMomentRow> bestJoy;
  final String currencyCode;
  final Locale locale;
  final ValueChanged<String>? onTap; // emits transactionId

  @override
  Widget build(BuildContext context) {
    final l10n = S.of(context);
    final formatter = FormatterService();

    return Card(
      shape: RoundedRectangleBorder(
        borderRadius: BorderRadius.circular(14),
        side: BorderSide(color: AppColors.soul.withValues(alpha: 0.20)),
      ),
      color: AppColors.soul.withValues(alpha: 0.08),
      child: Padding(
        padding: const EdgeInsets.all(14),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            Text(
              l10n.analyticsCardTitleBestJoy,
              style: AppTextStyles.titleLarge.copyWith(color: AppColors.soul),
            ),
            const SizedBox(height: 8),
            switch (bestJoy) {
              Empty() => _buildEmpty(context, l10n),
              Value(:final data) when (data as BestJoyMomentRow).soulSatisfaction <= 2 =>
                _buildEmpty(context, l10n),
              Value(:final data) =>
                _buildValue(context, l10n, formatter, data as BestJoyMomentRow),
            },
          ],
        ),
      ),
    );
  }

  Widget _buildEmpty(BuildContext context, S l10n) {
    return Text(
      l10n.analyticsCardEmptyBestJoy,
      style: AppTextStyles.bodyMedium.copyWith(
        color: context.wmTextSecondary,
      ),
    );
  }

  Widget _buildValue(
    BuildContext context,
    S l10n,
    FormatterService formatter,
    BestJoyMomentRow row,
  ) {
    final categoryName = row.categoryId; // TODO: CategoryLocalizationService.resolveFromId(row.categoryId, locale)
    final dateLabel = DateFormatter.formatShortMonthDay(row.timestamp, locale);
    final amountText = formatter.formatCurrency(row.amount, currencyCode, locale);

    return Semantics(
      // T-Information-1 mitigation: category + score + amount only.
      label: '$categoryName ${row.soulSatisfaction}/10 $amountText $dateLabel',
      child: GestureDetector(
        onTap: onTap == null ? null : () => onTap!(row.transactionId),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            Text(
              l10n.analyticsCardBestJoyBig(categoryName, dateLabel),
              style: AppTextStyles.bodyMedium.copyWith(
                color: context.wmTextPrimary,
                fontWeight: FontWeight.w700,
              ),
            ),
            const SizedBox(height: 4),
            Text(
              l10n.analyticsCardSmallBestJoy(amountText, row.soulSatisfaction),
              style: AppTextStyles.caption.copyWith(
                color: context.wmTextSecondary,
              ),
            ),
          ],
        ),
      ),
    );
  }
}

/// STATSUI-02 — Family Insight Card (物語 group · 家族 card, group mode only).
/// D-13 binding: rings → sentence-form rendering of Phase 9 family contracts.
/// D-14 binding: SEPARATE widget from HomeHeroCard's _legendGroup; this is
/// presentational sentence-form, not a ring re-render.
/// Anti-leaderboard: consumes ONLY familyHighlightsSum (int aggregate) +
/// sharedJoyInsight (3-tuple) — NO per-member fields (Phase 9 D-08
/// compile-time enforced; FamilyHappiness shape is `int`, not `Map<MemberId, int>`).
/// Pitfall 6 + Phase 10 D-08: render iff isGroupMode && shadowBooks.isNotEmpty.
class FamilyInsightCard extends StatelessWidget {
  const FamilyInsightCard({
    super.key,
    required this.family,
    required this.isGroupMode,
    required this.shadowBooks,
    required this.locale,
  });

  final FamilyHappiness? family;
  final bool isGroupMode;
  final List<dynamic>? shadowBooks; // ShadowBookInfo — opaque here; only nullness/empty checked
  final Locale locale;

  @override
  Widget build(BuildContext context) {
    final showFamily = isGroupMode && (shadowBooks?.isNotEmpty ?? false);
    if (!showFamily) return const SizedBox.shrink();

    final l10n = S.of(context);
    return Card(
      shape: RoundedRectangleBorder(
        borderRadius: BorderRadius.circular(14),
        side: BorderSide(color: AppColors.olive.withValues(alpha: 0.30)),
      ),
      // D-17 ochre fill (closest project token to #FFF7E6 — olive at low alpha).
      color: AppColors.olive.withValues(alpha: 0.10),
      child: Padding(
        padding: const EdgeInsets.all(14),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            Text(
              l10n.analyticsCardTitleFamilyInsight,
              style: AppTextStyles.titleLarge.copyWith(color: AppColors.olive),
            ),
            const SizedBox(height: 8),
            _highlightsLine(context, l10n),
            const SizedBox(height: 6),
            _sharedJoyLine(context, l10n),
          ],
        ),
      ),
    );
  }

  Widget _highlightsLine(BuildContext context, S l10n) {
    final highlightsResult = family?.familyHighlightsSum;
    final text = switch (highlightsResult) {
      null || Empty() => '—',
      Value(:final data) =>
        l10n.analyticsFamilyHighlightsSentence(data as int),
    };
    return Text(
      text,
      style: AppTextStyles.bodyMedium.copyWith(color: context.wmTextPrimary),
    );
  }

  Widget _sharedJoyLine(BuildContext context, S l10n) {
    final insightResult = family?.sharedJoyInsight;
    final text = switch (insightResult) {
      null || Empty() => l10n.analyticsFamilyEmpty,
      Value(:final data) => l10n.analyticsFamilySharedJoySentence(
          (data as SharedJoyInsight).categoryId,
          data.totalCount,
          data.avgSatisfaction.toStringAsFixed(1),
        ),
    };
    return Text(
      text,
      style: AppTextStyles.bodyMedium.copyWith(color: context.wmTextPrimary),
    );
  }
}
